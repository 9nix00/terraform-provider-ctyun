  name        = "tf-subnet-for-vpc"

  description = "terraform测试使用"
  dns = [
    "114.114.114.114",
    "8.8.8.8",
  ]

  vpc_id      = ctyun_vpc.vpc_test.id

}

  name       = "tf-bandwidth-for-vpc"
  bandwidth  = 5
  cycle_type = "on_demand"
}

resource "ctyun_eip" "eip_test" {
  name                = "tf-eip-for-vpc"
  bandwidth           = 5
  cycle_type          = "on_demand"
  demand_billing_type = "upflowc"
}

data "ctyun_images" "image_test" {
  name       = "CentOS Linux 8.4"
  visibility = "public"
  page_no    = 1
  page_size  = 10
}

  cpu    = 2

  arch   = "x86"
  series = "C"
  type   = "CPU_C7"
}

resource "ctyun_ecs" "ecs_test" {
  instance_name    = "tf-ecs-for-vpc"
  display_name     = "tf-ecs-for-vpc"
  flavor_id        = data.ctyun_ecs_flavors.ecs_flavor_test.flavors[0].id
  image_id         = data.ctyun_images.image_test.images[0].id
  system_disk_type = "sata"
  system_disk_size = 40
  vpc_id           = ctyun_vpc.vpc_test.id
  password         = var.password

  vpc_id     = ctyun_subnet.subnet_test.vpc_id

resource "ctyun_vip_association" "example" {
  vip_id               = ctyun_vip.example.id
  resource_type        = "VM"
  network_interface_id = ctyun_port.port_test.id

variable "password" {
  type      = string

## Schema

### Required

- `resource_type` (String) 绑定的实例类型，VM 表示虚拟机ECS, PM 表示裸金属, NETWORK 表示弹性 IP
- `vip_id` (String) 高可用虚IP的ID

### Optional

- `floating_id` (String) 弹性IP ID，当 resource_type 为 NETWORK 时，必填
- `instance_id` (String) ECS示例ID，当 resource_type 为 VM / PM 时，必填
- `network_interface_id` (String) 虚拟网卡ID, 该网卡属于instance_id, 当 resource_type 为 VM / PM 时，必填
- `region_id` (String) 资源池ID，如果不填则默认使用provider ctyun中的region_id或环境变量中的CTYUN_REGION_ID

### Read-Only

- `id` (String) 资源唯一标识，格式为 region_id:vip_id
